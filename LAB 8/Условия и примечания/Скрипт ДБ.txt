CREATE DATABASE University;
GO

USE University;
GO

CREATE TABLE UNIVERSITY (
    UNIV_ID BIGINT NOT NULL PRIMARY KEY,
    UNIV_NAME NVARCHAR(50) NULL,
    RATING INT NULL,
    CITY NVARCHAR(50) NULL
);
GO

CREATE TABLE STUDENT (
    STUDENT_ID INT NOT NULL PRIMARY KEY,
    SURNAME NVARCHAR(50) NULL,
    NAME NVARCHAR(50) NULL,
    STIPEND FLOAT NULL,
    KURS INT NULL,
    CITY VARCHAR(50) NULL,
    BIRTHDAY DATETIME NULL,
    UNIV_ID INT NULL
);
GO

CREATE TABLE LECTURES (
    LECTURER_ID INT NOT NULL PRIMARY KEY,
    SURNAME NVARCHAR(50) NULL,
    NAME NVARCHAR(50) NULL,
    CITY NVARCHAR(50) NULL,
    UNIV_ID INT NULL
);
GO

CREATE TABLE SUBJECT (
    SUBJ_ID INT NOT NULL PRIMARY KEY,
    SUBJ_NAME NVARCHAR(50) NULL,
    HOUR INT NULL,
    SEMESTER INT NULL
);
GO

CREATE TABLE EXAM_MARKS (
    EXAM_ID INT NOT NULL PRIMARY KEY,
    STUDENT_ID INT NULL,
    SUBJ_ID INT NULL,
    MARK INT NULL,
    EXAM_DATE DATETIME NULL
);
GO

ALTER TABLE STUDENT
ADD CONSTRAINT FK_STUDENT_UNIVERSITY
FOREIGN KEY (UNIV_ID) REFERENCES UNIVERSITY(UNIV_ID);
GO

ALTER TABLE LECTURES
ADD CONSTRAINT FK_LECTURES_UNIVERSITY
FOREIGN KEY (UNIV_ID) REFERENCES UNIVERSITY(UNIV_ID);
GO

ALTER TABLE EXAM_MARKS
ADD CONSTRAINT FK_EXAMMARKS_STUDENT
FOREIGN KEY (STUDENT_ID) REFERENCES STUDENT(STUDENT_ID);
GO

ALTER TABLE EXAM_MARKS
ADD CONSTRAINT FK_EXAMMARKS_SUBJECT
FOREIGN KEY (SUBJ_ID) REFERENCES SUBJECT(SUBJ_ID);
GO
